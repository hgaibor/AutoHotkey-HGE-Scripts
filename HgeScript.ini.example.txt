[CreateOpenFolder]
	# Variables listed here will be used by the function ProcessFolderSlot()
	# ProcessFolderSlot() will wait for a single digit key press defining the slot to use
	# Then it will call CreateOpenFolder() function to open predefined folder paths or create new folder inside of them: 
	# Current function supports ten slots with a single digit [0-9]
	# No need to create each of the 10 slots, however if set up one, make sure to fill the three required variables
	
	# VARIABLES DESCRIPTION
	## BaseFolderPath1= --> Base folder path to open or create folders into (no need to end with back slash \)
	## LocationName1= --> Descriptive name for location (not filling it will show the name as ERROR)
	## FolderOperation1= --> Current supported operations are [Open_Create|Open_Folder]
	### [Open_Create]: Will display an input box to enter the name of the folder to create, or open if already present
	### 							 No validation against invalid characters is made, so BE WARNED
	### [Open_Folder]: Will open the path defined on the current slot

	BaseFolderPath1="C:\Users\MyDocuments"
	LocationName1="Documents"
	FolderOperation1="Open_Create"

[OpenWebSiteWithInput]
	MyWebsite-SiteName="MyWebsite item"
	# To avoid duplicating browser launch command info, website open command will use one of the declared browser configs available at the '[Defined browsers]' section
	MyWebsite-BrowserProfile="DefaultHge"

	# Options are "New_Tab" and "New_Window", name is self explanatory,
	# 	but it will use the new tab or window commands defined on '[Defined browsers]' as the command would not be the same for different browsers
	MyWebsite-OpenTarget="New_Window"
	# MyWebsite-OpenTarget="New_Tab"

	# Use "DO_NOT_SEARCH" and "DO_NOT_GROUP" on these options to override searching and focus of current similar existing windows. 
	# 	This is only valid if the WebSite page title contains a similar set of characters that can allow to search and group pages.
	#		Also, this will only work if the selected tab with the page is the currently visible tab on a browser window.
	MyWebsite-AhkSearchWindowTitle="DO_NOT_SEARCH"
	MyWebsite-AhkGroupName="DO_NOT_GROUP"

	# BaseUrl --> URL that will be opened, you can put up to 10 placeholders that will be replaced with the inputs requested
	MyWebsite-BaseUrl="https://mycoolwebsite.com/index.php?/View/{REPLACE_INPUT1}"
	# Available options are "Open_Website" and "Copy_URL"
	MyWebsite-OpenWebSiteOperation="Open_Website"
	# MyWebsite-OpenWebSiteOperation="Copy_URL"
	# Enter up to 10 possible inputs, they need to be continous starting from 1, follow similar syntax: 
		# NameInput1 --> Informational name to identify the input that will be requested on this website open command
		# RegExDeleteFromInput1 --> Sanitize regex to remove unwanted characters from inputX, RegEx may start i) for case insensitive
		# 													More info at: https://www.autohotkey.com/docs/misc/RegEx-QuickRef.htm#Options
		# RegExInsertInput1  --> field from the "XXX-BaseUrl" field, where the entered input will be placed.
		# 											 For practical purposes it's named {REPLACE_INPUTX} but you may use any placeholder, as long as it is not part of the URL
		# 											 as long as it is not part of the URL and it has the same name here than on the URL
	MyWebsite-NameInput1="Item ID"
	MyWebsite-RegExDeleteFromInput1="i)[^\d]"
	MyWebsite-RegExInsertInput1="i)\{REPLACE_INPUT1\}"
